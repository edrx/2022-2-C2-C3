 (f·g)(x) &=& f(x)·g(x) \\

 (Df)(x) &=& \ddx f(x) \\

       D &=& λf.λx.\ddx f(x) \\
 \\
 D(g+h) &=& (λf.λx.\ddx f(x))(g+h) \\
        &=& λx.\ddx (g+h)(x) \\
        &=& λx.\ddx (g(x)+h(x)) \\
        &=& λx.\ddx g(x) + \ddx h(x)) \\
        &=& (λx.\ddx g(x)) + (λx.\ddx h(x))) \\
        &=& Dg + Dh \\
